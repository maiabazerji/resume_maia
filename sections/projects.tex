\mainsection{Projets \& Comp\'etitions}

\projectentry{RoboCar -- Epitech (Juin 2025)}{%
D\'eveloppement de l'IA pour navigation autonome et prise de d\'ecision temps r\'eel (\textbf{Computer Vision, Deep Learning}).}

\projectentry{Catalog Intelligence \& Anomaly Detection}{%
Syst\`eme de d\'etection de fraudes et comportements suspects en temps r\'eel. Extraction NER multimodale pour l'onboarding produit (\textbf{CamemBERT, Whisper, Scikit-learn}).}

\projectentry{Taxi Driver -- Deep Reinforcement Learning}{%
Agents autonomes sur Frozen Lake. Benchmarking Q-Learning vs SARSA et impl\'ementation de \textbf{Deep Q-Networks (DQN)} avec PyTorch et Gymnasium.}

\projectentry{Sentiment Analysis Pipeline}{%
Fine-tuning de mod\`eles Transformers pour la priorisation automatique de messages positifs et n\'egatifs (\textbf{Hugging Face, FastAPI}).}

\projectentry{Computer Vision Quality Control}{%
Classification d'images avec optimisation des DataLoaders pour le traitement de flux massifs en production (\textbf{PyTorch, CNN, OpenCV}).}

\projectentry{KubeQuest -- Cloud-Native Infrastructure}{%
D\'eploiement d'un cluster Kubernetes avec stack d'observabilit\'e compl\`ete. Automatisation GitOps et s\'ecurisation via OPA et OAuth2 (\textbf{Terraform, Helm, Prometheus}).}
